# Step 1: Initialize Git in your project folder
git init

# Step 2: Add all files to Git
git add .

# Step 3: Commit your files
git commit -m "Full Bank Wallet app - initial commit"

# Step 4: Rename the default branch to main
git branch -M main

# Step 5: Add your GitHub repository as remote
git remote add origin https://github.com/YOUR_USERNAME/bank-wallet.git

# Step 6: Push all files to GitHub
git push -u origin main
